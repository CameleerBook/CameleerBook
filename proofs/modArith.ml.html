module ModularArithmetic (M: sig val m:int end) = struct

	(*@ lemma modular_overflow: forall a b m.
		0 <= a < m && 0 <= b < m -> 
		m <= a+b < 2*m -> 
		0 <= a+b-m < m -> 
		mod (a+b) m = a+b-m *)

	(*@ lemma modular_underflow: forall a b m.
		0 <= a < m && 0 <= b < m && a < b -> 
		-m < a-b < 0 -> 
		0 < a-b+m < m -> 
		mod (a-b) m = a-b+m *)

	(* lemma modular_mult_overflow: forall a b m.
		0 <= a < m && 0 <= b < m -> 
		0 <= a*b < m*m -> 
		0 <= div (a*b) m < m -> 
		mod (a*b) m = div (a*b) m *)

	(* lemma mult_interval: forall a b m.
		0 <= a < m && 0 <= b < m -> 0 <= a*b < m*m *)

	(* lemma modular_division: forall a b m.
		0 <= a < m && 0 < b < m -> 
		0 <= div a b < m -> 
		mod (div a b) m = div a b *)

	let modulo (x: int) : int =
		let r = x mod M.m in if r < 0 then r + M.m else r
	(*@ r = modulo x
		requires M.m > 0
		ensures 0 <= r < M.m
		ensures mod x M.m < 0 -> r = M.m + mod x M.m
		ensures mod x M.m >= 0 -> r = mod x M.m *)

	let add (a: int) (b: int) : int =
		let res = a + b in if res >= M.m then res - M.m else res	
	(*@ r = add a b
		requires 0 <= a < M.m
		requires 0 <= b < M.m
		requires M.m > 0
		ensures 0 <= r < M.m 
		ensures r = mod (a+b) M.m *)

	let sub (a: int) (b: int) : int =
		let res = a - b in if res < 0 then res + M.m else res	
	(*@ r = sub a b
		requires 0 <= a < M.m
		requires 0 <= b < M.m
		requires M.m > 0
		ensures 0 <= r < M.m 
		ensures r = mod (a-b) M.m *)

	let mul (a: int) (b: int) : int =
		let res = a * b in if res >= M.m then res - M.m * (res/M.m) else res
	(*@ r = mul a b
		requires 0 <= a < M.m
		requires 0 <= b < M.m
		requires M.m > 0
		ensures 0 <= r < M.m 
		ensures r = mod (a*b) M.m *)

	let division (a: int) (b: int) : int = a/b
	(*@ r = division a b
		requires 0 <= a < M.m
		requires 0 < b < M.m
		requires M.m > 0
		ensures 0 <= r < M.m 
		ensures r = mod (div a b) M.m *)

end
