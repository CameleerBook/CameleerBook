Ternary Search (With Exceptions)

(* Implementation adapted from https://www.geeksforgeeks.org/ternary-search/ *)
(* Specification by Pedro Gasparinho, advised by MÃ¡rio Pereira *)

(*@ predicate sorted (a: int array) = forall i j:int. 
    0 <= i <= j < Array.length a -> a.(i) <= a.(j) *)

let ternarySearch a v =
  let exception Break of int in
  try
    let l = ref 0 in 
    let u = ref (Array.length a - 1) in
    while !l <= !u do
    (*@ invariant 0 <= !l
        invariant !u < Array.length a
        invariant forall k. 0 <= k < !l \/ !u < k < Array.length a -> a.(k) <> v
        variant !u - !l *)
      let m1 = !l + (!u - !l)/3 in
      let m2 = !u - (!u - !l)/3 in
      if (a.(m1) = v) then raise (Break m1)
      else if (a.(m2) = v) then raise (Break m2)
      else if (v < a.(m1)) then u := m1 - 1
      else if (v > a.(m2)) then l := m2 + 1
      else begin l := m1 + 1; u := m2 - 1 end
    done;
    raise Not_found
  with Break i -> i
(*@ r = ternarySearch a v
    requires sorted a
    ensures a.(r) = v
    raises Not_found -> forall k. 0 <= k < Array.length a -> a.(k) <> v *)
