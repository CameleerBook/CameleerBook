let seconds_to_hours s =
  let h = s / 3600 in
  let r = s mod 3600 in
  let m = r / 60 in
  let s' = r mod 60 in
  (h, m, s')
(*@ h, m, s' = seconds_to_hours s
    requires s >= 0
    ensures 3600*h + 60*m + s' = s *)
