(* Note: Proof is not 'exhaustive' on purpose *)

let rec count x l =
  match l with
  | [] -> 0
  | h::t -> (if x = h then 1 else 0) + count x t

    variant l
    ensures 0 <= r <= List.length l
    ensures r > 0 -> List.mem x l
    ensures r = 0 -> not List.mem x l *)
