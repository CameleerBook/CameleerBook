module type Cmp = sig
	type t

	val eq: t -> t -> bool [@@logic]
	(*@ b = eq x y
		ensures b <-> x = y *)

	(*@ function le: t -> t -> bool *)

	(*@ axiom reflexive : forall x. le x x *)
	(*@ axiom total 		: forall x y. le x y \/ le y x *)
	(*@ axiom transitive: forall x y z. le x y -> le y z -> le x z *)

	val leq: t -> t -> bool
	(*@ b = leq x y
			ensures b <-> le x y *)
end

module InsertionSort (E: Cmp) = struct
	type elt = E.t

	(*@ function rec occ (v: elt) (l: elt list) : int = 
		match l with
		| [] -> 0
		| x::ls -> (if E.eq x v then 1 else 0) + (occ v ls) *)

	(*@ lemma occ_nonneg: forall x l.
		occ x l >= 0 *)
		
	(*@ lemma occ_mem: forall x l.
		occ x l > 0 <-> List.mem x l *)
		
	(*@ lemma occ_append: forall x l1 l2.
		    occ x (l1 @ l2) = occ x l1 + occ x l2 *)

	(*@ predicate permut (l1 l2: elt list) =
		forall x. occ x l1 = occ x l2 *)
		
	(*@ lemma permut_len: 
		forall l1, l2: elt list.
		permut l1 l2 -> (List.length l1 = List.length l2) *)

	(*@ predicate rec sorted (l: elt list) =
		match l with
		| [] | _::[] -> true
		| x::(y::ls) -> E.le x y && sorted (y::ls) *)
	(*@ variant l *)

	(*@ lemma sorted_append :
		forall l1, l2: elt list.
		(sorted l1 /\ sorted l2 /\
		(forall x y. List.mem x l1 -> List.mem y l2 -> E.le x y)) <->
		sorted (l1 @ l2) *)
		
	(*@ lemma sorted_cons :
		forall x:elt, l: elt list.
		(sorted l /\
		(forall e. List.mem e l -> E.le x e)) <->
		sorted (x::l) *)

	let rec insert x l =
		match l with
		| [] -> [x]
		| y::ls -> 
			if E.leq x y then x::y::ls
			else y::(insert x ls)
	(*@ r = insert x l
			requires sorted l
			ensures sorted r
			ensures permut r (x::l)
			variant l *)

	let rec insertion_sort l =
		match l with
		| [] -> []
		| [x] -> [x] 
		| x::ls -> insert x (insertion_sort ls)
	(*@ r = insertion_sort l
			ensures sorted r
			ensures permut r l
			variant l *)
			  
end
