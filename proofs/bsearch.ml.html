(*@ predicate sorted (a: int array) =
  		forall i j:int. 0 <= i <= j < Array.length a -> a.(i) <= a.(j)*)

let binary_search (a : int array) (v: int) =
    let l = ref 0 in
    let u = ref (Array.length a - 1) in
    let res = ref (-1) in
    while !res < 0 && !l <= !u do
    	(*@ invariant 0 <= !l
        	invariant !u < Array.length a
        	invariant -1 <= !res < Array.length a
        	invariant !res >= 0 -> a.(!res) = v
        	invariant !res = -1 -> forall k. 0 <= k < !l \/ !u < k < Array.length a -> a.(k) <> v
        	variant !u - !l - !res - 1 *)
        let m = (!u + !l)/ 2 in
        if a.(m) < v then l := m+1
        else if a.(m) > v then u := m-1
        else res := m
      done;
    !res
