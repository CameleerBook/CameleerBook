let gcd (x:int) (y:int) = 
	let xs = ref x in 
	let ys = ref y in
  	let[@ghost] a = ref 1 in
  	let[@ghost] b = ref 0 in
  	let[@ghost] c = ref 0 in
  	let[@ghost] d = ref 1 in
  	while !ys > 0 do
    	(*@ invariant !xs >= 0
    		invariant !ys >= 0
    		invariant !xs = !a * x + !b * y
    		invariant !ys = !c * x + !d * y
    		variant !ys *)
    	let r = !xs mod !ys in 
    	let[@ghost] q = !xs / !ys in
    	xs := !ys; 
    	ys := r;
    	let e = !a in
    	let f = !b in
    	a := !c;
    	b := !d;
    	c := e - q * !c;
    	d := f - q * !d;
  	done;
  	!xs
(*@ r = gcd x y
   	requires x >= 0
  	requires y >= 0
  	ensures exists a,b. r = a*x+b*y
*)
