module type Cmp = sig
	type t

	val eq: t -> t -> bool [@@logic]
	(*@ b = eq x y
		ensures b <-> x = y *)

	(*@ function le: t -> t -> bool *)

	(*@ axiom reflexive 		: forall x. le x x *)
	(*@ axiom total 				: forall x y. le x y \/ le y x *)
	(*@ axiom transitive_le	: forall x y z. le x y -> le y z -> le x z *)

	val leq: t -> t -> bool
	(*@ b = leq x y
			ensures b <-> le x y *)

	(*@ function g: t -> t -> bool *)

	(*@ axiom inverse				: forall x y. g x y <-> not le x y *)
	(*@ axiom transitive_g	: forall x y z. g x y -> g y z -> g x z *)
	
end

module QuickSort (E: Cmp) = struct
	type elt = E.t

	(*@ function rec occ (v: elt) (l: elt list) : int = 
		match l with
		| [] -> 0
		| x::ls -> (if E.eq x v then 1 else 0) + (occ v ls) *)

	(*@ lemma occ_nonneg: forall x l.
		occ x l >= 0 *)
		
	(*@ lemma occ_mem: forall x l.
		occ x l > 0 <-> List.mem x l *)
		
	(*@ lemma occ_append: forall x l1 l2.
		    occ x (l1 @ l2) = occ x l1 + occ x l2 *)

	(*@ predicate permut (l1 l2: elt list) =
		forall x. occ x l1 = occ x l2 *)
		
	(*@ lemma permut_len: 
		forall l1, l2: elt list.
		permut l1 l2 -> (List.length l1 = List.length l2) *)

	(*@ predicate rec sorted (l: elt list) =
		match l with
		| [] | _::[] -> true
		| x::(y::ls) -> E.le x y && sorted (y::ls) *)
	(*@ variant l *)

	(*@ lemma sorted_append :
		forall l1, l2: elt list.
		(sorted l1 /\ sorted l2 /\
		(forall x y. List.mem x l1 -> List.mem y l2 -> E.le x y)) <->
		sorted (l1 @ l2) *)
		
	(*@ lemma sorted_cons :
		forall x:elt, l: elt list.
		(sorted l /\
		(forall e. List.mem e l -> E.le x e)) <->
		sorted (x::l) *)

	let rec quick_aux p = function
		| [] -> ([], [])
		| x :: r ->
			let ll, lr = quick_aux p r in
			if E.leq x p then x :: ll, lr
			else ll, x :: lr
	(*@ ll, lr = quick_aux p l
			variant l
			ensures forall x. List.mem x ll -> E.le x p
			ensures forall x. List.mem x lr -> E.g x p
			ensures permut l (ll ++ lr) *)

	let rec quick_sort l =
		match l with
		| [] -> []
		| [x] -> [x]
		| p :: ls ->
			let (left, right) = (quick_aux p ls) in
			(quick_sort left) @  p :: quick_sort right
	(*@ r = quick_sort l
			ensures sorted r
			ensures permut r l
			variant List.length l *)
  
end
