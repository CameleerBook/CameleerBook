Ternary Search

(* Implementation adapted from https://www.geeksforgeeks.org/ternary-search/ *)
(* Specification by Pedro Gasparinho, advised by MÃ¡rio Pereira *)

(*@ predicate sorted (a: int array) =
  		forall i j:int. 0 <= i <= j < Array.length a -> a.(i) <= a.(j)*)

let ternarySearch a v =
  let l = ref 0 in 
  let u = ref (Array.length a - 1) in
  let r = ref (-1) in
  while !r < 0 && !l <= !u do
	(*@ invariant 0 <= !l
    	invariant !u < Array.length a
    	invariant -1 <= !r < Array.length a
    	invariant !r >= 0 -> a.(!r) = v
    	invariant !r = -1 -> forall k. 0 <= k < !l \/ !u < k < Array.length a -> a.(k) <> v
    	variant !u - !l - !r - 1 *)
    let mid1 = !l + (!u - !l)/3 in
    let mid2 = !u - (!u - !l)/3 in
    if (a.(mid1) = v) then r := mid1 
    else if (a.(mid2) = v) then r := mid2 
    else if (v < a.(mid1)) then u := mid1 - 1
    else if (v > a.(mid2)) then l := mid2 + 1
    else begin l := mid1 + 1; u := mid2 - 1 end
  done;
  !r
(*@ r = ternarySearch a v
		requires sorted a
  	ensures -1 <= r < Array.length a
  	ensures r >= 0 -> a.(r) = v
  	ensures r = -1 -> forall k. 0 <= k < Array.length a -> a.(k) <> v *)
