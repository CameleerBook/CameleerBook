module type Cmp = sig
	type t

	val eq: t -> t -> bool [@@logic]
	(*@ b = eq x y
		ensures b <-> x = y *)

	(*@ function le: t -> t -> bool *)

	(*@ axiom reflexive : forall x. le x x *)
	(*@ axiom total 		: forall x y. le x y \/ le y x *)
	(*@ axiom transitive: forall x y z. le x y -> le y z -> le x z *)

	val leq: t -> t -> bool
	(*@ b = leq x y
			ensures b <-> le x y *)
end

module SelectionSort (E: Cmp) = struct
	type elt = E.t

	(*@ function rec occ (v: elt) (l: elt list) : int = 
		match l with
		| [] -> 0
		| x::ls -> (if E.eq x v then 1 else 0) + (occ v ls) *)

	(*@ lemma occ_nonneg: forall x l.
		occ x l >= 0 *)
		
	(*@ lemma occ_mem: forall x l.
		occ x l > 0 <-> List.mem x l *)
		
	(*@ lemma occ_append: forall x l1 l2.
		    occ x (l1 @ l2) = occ x l1 + occ x l2 *)

	(*@ predicate permut (l1 l2: elt list) =
		forall x. occ x l1 = occ x l2 *)
		
	(*@ lemma permut_len: 
		forall l1, l2: elt list.
		permut l1 l2 -> (List.length l1 = List.length l2) *)

	(*@ predicate rec sorted (l: elt list) =
		match l with
		| [] | _::[] -> true
		| x::(y::ls) -> E.le x y && sorted (y::ls) *)
	(*@ variant l *)

	(*@ lemma sorted_append :
		forall l1, l2: elt list.
		(sorted l1 /\ sorted l2 /\
		(forall x y. List.mem x l1 -> List.mem y l2 -> E.le x y)) <->
		sorted (l1 @ l2) *)
		
	(*@ lemma sorted_cons :
		forall x:elt, l: elt list.
		(sorted l /\
		(forall e. List.mem e l -> E.le x e)) <->
		sorted (x::l) *)

	let rec selection_aux min = function
		| [] -> (min, [])
		| x :: r ->
			if E.leq min x then
				let (m, l) = selection_aux min r in
				m, x :: l
			else
				let (m, l) = selection_aux x r in
				m, min :: l
	(*@ m, r = selection_aux min l
			variant l
			ensures E.le m min
			ensures forall x. List.mem x r -> E.le m x
			ensures List.length r = List.length l
			ensures permut (m :: r) (min :: l) *)

	let rec selection_sort l =
		match l with
		| [] -> []
		| [x] -> [x]
		| x :: ls ->
			let m, r = selection_aux x ls in
			m :: (selection_sort r)
	(*@ r = selection_sort l
			ensures sorted r
			ensures permut r l
			variant List.length l *)
	
end
