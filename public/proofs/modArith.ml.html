	let[@logic] modulo (x: int) (m: int) : int =
		let r = x mod m in if r < 0 then r + m else r
	(*@ r = modulo x m
			requires m > 0
			ensures 0 <= r < m
			ensures mod x m < 0 -> r = m + mod x m
			ensures mod x m >= 0 -> r = mod x m *)

	let add (a: int) (b: int) : int =
		let res = a + b in if res >= M.m then res - M.m else res	
	(*@ r = add a b
			requires 0 <= a < M.m
			requires 0 <= b < M.m
			requires M.m > 0
			ensures 0 <= r < M.m 
			ensures r = mod (a+b) M.m *)

	let sub (a: int) (b: int) : int =
		let res = a - b in if res < 0 then res + M.m else res	
	(*@ r = sub a b
			requires 0 <= a < M.m
			requires 0 <= b < M.m
			requires M.m > 0
			ensures 0 <= r < M.m 
			ensures r = mod (a-b) M.m *)

	let mul (a: int) (b: int) : int =
		let res = a * b in if res >= M.m then res / M.m else res
	(*@ r = mul a b
			requires 0 <= a < M.m
			requires 0 <= b < M.m
			requires M.m > 0
			ensures 0 <= r < M.m 
			ensures r = mod (a*b) M.m *)

	let division (a: int) (b: int) : int = a/b

			requires 0 <= a < M.m

			ensures 0 <= r < M.m 
			ensures r = mod (div a b) M.m *)
