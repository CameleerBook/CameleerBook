(*@ function rec occ (v: int) (l: int list) : int =
	match l with
	| [] -> 0
	| x::ls -> (if x = v then 1 else 0) + (occ v ls) *)

(*@ lemma occ_nonneg: forall x l.
	occ x l >= 0 *)

(*@ lemma occ_mem: forall x l.
	occ x l > 0 <-> List.mem x l *)

(*@ lemma occ_append: forall x l1 l2.
      occ x (l1 @ l2) = occ x l1 + occ x l2 *)

(*@ predicate permut (l1 l2: int list) =
	forall x. occ x l1 = occ x l2 *)

(*@ lemma permut_len:
	forall l1, l2: int list.
	permut l1 l2 -> (List.length l1 = List.length l2) *)

(*@ predicate rec sorted (l: int list) =
	match l with
	| [] -> true
	| _::[] -> true
	| x::(y::ls) -> x <= y && sorted (y::ls) *)
(*@ variant l *)

(*@ lemma sorted_append :
	forall l1, l2: int list.
	(sorted l1 /\ sorted l2 /\
	(forall x y. List.mem x l1 -> List.mem y l2 -> x <= y)) <->
	sorted (l1 @ l2) *)

(*@ lemma sorted_cons :
	forall x:int, l: int list.
	(sorted l /\
	(forall e. List.mem e l -> x <= e)) <->
	sorted (x::l) *)

let rec insert x acc l =
	match l with
	| [] -> acc @ [x]
	| y::ls -> 
		if x <= y then acc @ x::l
		else insert x (acc @ [y]) ls
(*@ r = insert x acc l
	requires sorted acc
	requires forall k. List.mem k acc -> k <= x
	requires forall k y. List.mem k acc -> List.mem y l -> k <= y
	requires sorted l
	ensures sorted r
	ensures permut r (acc ++ (x::l))
	variant l *)

let rec insertion_sort l acc =
	match l with
	| [] -> acc 
	| x::ls -> insertion_sort ls (insert x [] acc)
(*@ r = insertion_sort l acc
	requires sorted acc
	ensures sorted r
	ensures permut r (l @ acc)
	variant l *)
