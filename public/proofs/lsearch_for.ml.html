Linear Search (For Loop)

let linear_search a v = 
  let res = ref (-1) in
  for i = 0 to Array.length a - 1 do
    (*@ invariant -1 <= !res < Array.length a
    	invariant !res >= 0 -> a.(!res) = v
    	invariant !res = -1 -> forall k. 0 <= k < i -> a.(k) <> v*)
    let t = a.(i) in
    if t = v then res := i
  done;
  !res
(*@ r = linear_search a v
	ensures -1 <= r < Array.length a
