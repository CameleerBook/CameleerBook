	if n > 0 then x * power x (n-1)
	else 1 *)	
(*@ requires n >= 0

(*@ lemma power_mult :
      forall x y n:int.
      n >= 0 -> power (x*y) n = power x n * power y n
*)

(*@ lemma power_even:
      forall x:int, n:int. n >= 0 /\ mod n 2 = 0 ->
      power x n = (power (x * x) (div n 2))
*)

(*@ lemma power_odd:
      forall x:int, n:int. n >= 0 /\ mod n 2 = 1 ->
      power x n = x * (power (x * x) (div n 2)) 
*)

let fastexp x n =
	let r = ref 1 in
	let p = ref x in
	let e = ref n in
	while !e > 0 do
	(*@ invariant !r * power !p !e = power x n
	    invariant !e >= 0
	    variant !e *)
		if !e mod 2 = 1 then r := !r * !p
		else assert (!e mod 2 = 0);
		p := !p * !p;
		e := !e / 2;
	done;
	!r
(*@ r = fastexp x n
	requires n >= 0
    ensures r = power x n *)
