let rec eudiv_aux (x: int) y q r =
  if r < y then (q, r)
  else eudiv_aux x y (q+1) (r-y)
(*@	(q', r') = eudiv_aux x y q r
    requires y > 0
    requires r >= 0
    requires x = y * q + r
    ensures 0 <= r' < y 
    ensures x = y * q' + r'
    variant r *)

let eudiv x y = eudiv_aux x y 0 x
(*@ (q, r) = eudiv x y
    requires x >= 0
    requires y > 0
    ensures x = y * q + r
