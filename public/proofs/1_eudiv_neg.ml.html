Euclidean Division That Supports Negative Numbers (Imperative)

(* Specification by Pedro Gasparinho, advised by Mário Pereira *)

let sign n =
	if n >= 0 then 1
	else -1
(*@ r = sign n
		ensures n >= 0 -> r = 1
		ensures n < 0 -> r = -1 *)
	
let euclidean_div x y =
	let r = ref x in
	let q = ref 0 in
	let sx = sign x and sy = sign y in
	while not (0 <= !r && !r < abs y) do
	(*@	invariant x = y * !q + !r
			invariant x >= 0 -> 0 <= !r
			invariant x < 0 -> !r < abs y
			variant !r * sx + abs y *)
		r := !r - sx *  abs y;
		q := !q + 1 * sy * sx;
	done;
	(!q, !r)
(*@	(q, r) = euclidean_div x y
    requires y <> 0
    ensures x = y * q + r
    ensures 0 <= r < abs y *)
