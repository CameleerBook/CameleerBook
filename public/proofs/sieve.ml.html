(* Implementation adapted from Learn Programming with OCaml *)

(*@ predicate prime (n: int) =
    n > 1 && forall x. 1 < x < n -> mod n x <> 0 *)

(*@ lemma not_prime:
    forall n x. 1 < x < n && mod n x = 0 <-> not prime n *)

(*@ lemma mod_squared:
    forall n. n >= 1 <-> mod (n*n) n = 0 *)

(*@ lemma sum_of_mult:
    forall m n. n >= 1 && m >= 1 && mod m n = 0 <-> mod (m+n) n = 0 *)

(*
invariant n*n <= !m
          invariant mod !m n = 0
          invariant prime n
          invariant forall k. n < k < !m && mod k n = 0 -> not prime k

*)

let sieve max =
  let isPrime = Array.make (max + 1) true in
  isPrime.(0) <- false;
  isPrime.(1) <- false;
  let limit = max / 2 in
  for n = 2 to limit do
  (*@ invariant forall k. 0 <= k <= n -> (isPrime.(k) = prime k) *)
    if isPrime.(n) then begin
      let m = ref (n*n) in
      while !m <= max do
      (*@ variant max - !m *)
        isPrime.(!m) <- false;
        m := !m + n
      done
    end
  done;
  isPrime
(*@ isPrime = sieve max
    requires max > 1
    ensures Array.length isPrime = max + 1
    ensures forall k. 0 <= k <= max -> (isPrime.(k) = prime k) *)
