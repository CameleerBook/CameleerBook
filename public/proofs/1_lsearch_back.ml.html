Linear Search (While Loop & Backwards Iteration)

let lsearch a v =
  let exception Break of int in
  try
    let i = ref (Array.length a - 1) in
    while !i >= 0 do
    (*@ invariant -1 <= !i < Array.length a
        invariant forall k. !i < k < Array.length a -> a.(k) <> v
        variant !i *)
      if v = a.(!i) then raise (Break !i)
      else i := !i - 1
    done;
    raise Not_found
  with Break i -> i

    ensures a.(r) = v
    raises Not_found -> forall k. 0 <= k < Array.length a -> a.(k) <> v *)
