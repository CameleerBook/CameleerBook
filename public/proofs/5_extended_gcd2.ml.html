Extended GCD (Imperative)

let gcd x y = 
	
	let xs = ref x and ys = ref y in
	let [@ghost] a = ref 1 and [@ghost] a_next = ref 0 in
	let [@ghost] b = ref 0 and [@ghost] b_next = ref 1 in

	while !ys > 0 do
	(*@ invariant !xs >= 0
		invariant !ys >= 0
		invariant !xs = !a * x + !b * y
		invariant !ys = !a_next * x + !b_next * y
		variant !ys *)
		let[@ghost] q = !xs / !ys in
		let r = !xs mod !ys in 
		xs := !ys; 
		ys := r;
		
		let a' = !a in 
		a := !a_next; 
		a_next := a' - q * !a_next;
		
		let b' = !b in 
		b := !b_next; 
		b_next := b' - q * !b_next;
	
	done;
	!xs
(*@ r = gcd x y
   	requires x >= 0
  	requires y >= 0
  	ensures exists a,b. r = a*x+b*y *)
