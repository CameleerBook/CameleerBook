let linear_search a v = 
  let exception Break of int in 
  try
    for i = 0 to Array.length a - 1 do
    (*@ invariant forall k. 0 <= k < i -> a.(k) <> v *)
      if a.(i) = v then raise (Break i)
    done;
    -1
  with Break i -> i
(*@ r = linear_search a v
  	ensures r >= 0 -> a.(r) = v
  	ensures r = -1 -> forall k. 0 <= k < Array.length a -> a.(k) <> v *)
